#include <bits/stdc++.h>

using namespace std;

template <class T>
int manacher (const T & v) {
    int n = v.size() * 2 + 1;
    T arr(n) , res(n , 0);
    arr[0] = -1 , arr.back() = -3;

    for (int i = 1 ;i < n - 1;i++)
        arr[i] = i&1 ? v[i / 2] : -1;

    int c = 0 , r = 0 , mx = 1;
    for (int i = 1 ;i < arr.size() - 1;i++) {
        int mirror = 2 * c - i;

        if (i < r)
            res[i] = min(res[mirror] , r - i);

        while (arr[i - res[i] - 1] == arr[i + res[i] + 1])
            res[i]++;

        if (i + res[i] > r) {
            c = i;
            r = i + res[i];
        }

        if (i&1)
            mx = max(mx , res[i] + (res[i] % 2 == 0));
        else
            mx = max(mx , res[i] + (res[i]&1));
    }

    return mx;
}
