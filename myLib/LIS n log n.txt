#include <bits/stdc++.h>

using namespace std;

int get(vector<int> &v, int l, int r, int key) {
    while (r - l > 1) {
        int m = l + (r - l) / 2;
        if (v[m] >= key)
            r = m;
        else
            l = m;
    }
    return r;
}

int lis(vector<int> &v) {
    if (v.size() == 0)
        return 0;
    vector<int> t(v.size(), 0);
    int length = 1;
    t[0] = v[0];
    for (int i = 1; i < v.size(); i++) {
        if (v[i] < t[0])
            t[0] = v[i];
        else if (v[i] > t[length - 1])
            t[length++] = v[i];
        else
            t[get(t, -1, length - 1, v[i])] = v[i];
    }

    return length;
}
